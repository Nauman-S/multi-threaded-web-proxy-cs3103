   call second;
   a = 4;
   call fourth;
}

procedure second {
   a = 1;
   while (a > 2) {
      call third;
   }
}

procedure third {
   a = 1;
}

   if (x > 2) then {
      x = 5;
   } else {
      call fifth;
   }

   if (x > 2) then {
       x = 1;
       while (x > 0) {
           call third;
       }
   } else {
       x = 2;
   }
