    x = 1;
    y = 1;
    while (x == 1) {
        a = x;
        read y;
        a = y;
        y = 3 + a + y; 
    }

}

        c = a + b;
        read c;

    }
    x = 1;

procedure NestedWhile {
    x = 1;
    while (x >= 2) {
        a = 3 + x;
        y = 3 + a + x;
        while (x == 3) {
            z = 3 + a + y;
            read a;
            c = a + y;
        }
        z = 3 + a;
    }
    z = 3 + x;
}

procedure DoubleNestedWhile {
    x = 2;
